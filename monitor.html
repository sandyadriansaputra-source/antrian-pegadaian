const scriptURL = 'https://script.google.com/macros/s/AKfycbxr54D2wJB7Lp9Nx9au126J82orMSzEqOYjPAFc1XnprHUlmQ9h47eg3mzYkP0fXayr/exec'; let nomorTerakhir = ""; function updateMonitor() { console.log("Mengecek data antrian..."); fetch(scriptURL) .then(res => res.json()) .then(data => { // Balik data agar yang terbaru ada di atas let dataTerbaru = data.reverse(); // Cari baris yang statusnya "Dipanggil" let antrian = dataTerbaru.find(row => row[2] === "Dipanggil"); if (antrian) { console.log("Ditemukan antrian dipanggil:", antrian[1]); if (antrian[1] !== nomorTerakhir) { nomorTerakhir = antrian[1]; document.getElementById('nomorDisplay').innerText = antrian[1]; document.getElementById('namaDisplay').innerText = antrian[0]; panggilSuara(antrian[1]); } } else { console.log("Tidak ada antrian dengan status 'Dipanggil'"); } }) .catch(err => { console.error("Gagal update monitor:", err); document.getElementById('namaDisplay').innerText = "Error: Cek Koneksi/URL"; }); } function panggilSuara(nomor) { let msg = new SpeechSynthesisUtterance(`Nomor antrian ${nomor}, silakan menuju loket`); msg.lang = 'id-ID'; window.speechSynthesis.speak(msg); } // Cek setiap 3 detik setInterval(updateMonitor, 3000);
